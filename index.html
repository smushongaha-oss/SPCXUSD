TradingView Pine Script for GMX Pre-IPO SpaceX Synthetic

TradingView Pine Script v5 

//@version=5
indicator("SPCXUSD - GMX Synthetic Monitor", shorttitle="SPCXUSD GMX", overlay=true, max_bars_back=1000)

// === INPUTS ===
showEMA     = input.bool(true,  "Show EMA 50/200")
showFunding = input.bool(true,  "Show Estimated Funding Rate")
leverage    = input.int(10,     "Max Leverage", minval=1, maxval=50)

// === PRICE DATA ===
price = close
ema50 = ta.ema(price, 50)
ema200 = ta.ema(price, 200)

// === MOMENTUM & STRENGTH ===
rsi = ta.rsi(price, 14)
mom = ta.mom(price, 10)

// === BACKGROUND COLOR ===
bgColor = price > ema200 ? color.new(#22c55e, 90) : color.new(#ef4444, 90)

// === PLOTS ===
plot(price,      title="SPCXUSD Price", color=color.white, linewidth=3)
plot(showEMA ? ema50  : na, title="EMA 50",  color=color.orange, linewidth=2)
plot(showEMA ? ema200 : na, title="EMA 200", color=color.blue,   linewidth=2)

// Estimated Funding Rate (GMX-style)
estFunding = (price - price[1]) / price * 100 * 8
plot(showFunding ? estFunding : na, title="Est. Funding %", color=color.purple, style=plot.style_columns, linewidth=3)

// === SHAPES & LABELS ===
bgcolor(bgColor, title="Trend Background")

plotshape(rsi > 70,  title="Overbought",  location=location.abovebar, color=color.red,   style=shape.triangledown, size=size.small)
plotshape(rsi < 30,  title="Oversold",    location=location.belowbar, color=color.green, style=shape.triangleup,   size=size.small)

// Alerts
alertcondition(ta.crossover(price, ema50),  "Bullish Cross",  "SPCXUSD crossed above EMA50")
alertcondition(ta.crossunder(price, ema200), "Bearish Cross", "SPCXUSD crossed below EMA200")

// Info Table
var table info = table.new(position.top_right, 2, 6, bgcolor=color.new(#1f2937, 80))
if bar_index % 10 == 0
    table.cell(info, 0, 0, "SPCXUSD", text_color=color.white)
    table.cell(info, 1, 0, str.tostring(price, "#.##"), text_color=color.white)
    table.cell(info, 0, 1, "RSI", text_color=color.white)
    table.cell(info, 1, 1, str.tostring(rsi, "#.##"), text_color=rsi>70?color.red:rsi<30?color.green:color.white)
    table.cell(info, 0, 2, "Est Funding", text_color=color.white)
    table.cell(info, 1, 2, str.tostring(estFunding, "#.##") + "%", text_color=color.purple)
    table.cell(info, 0, 3, "Trend", text_color=color.white)
    table.cell(info, 1, 3, price > ema200 ? "BULLISH" : "BEARISH", text_color=price>ema200?color.green:color.red)
